    def check_process_guards(self) -> None:
        if not self.guards_enabled:
            return
        if self.din(IN_DROPPED_LOAD):
            raise RuntimeError("DROPPED_LOAD: drop-zone sensor triggered")
        if (
            self.cargo_expected
            and self.guard_mode is GuardMode.CARRIAGE
            and not self.din(IN_CARGO_CHECKPOINT)
        ):
            raise RuntimeError("CARGO_LOST: carriage cargo signal dropped")

    def begin_loaded_transit(self) -> None:
        self.check_process_guards()
        if self.guards_enabled and not self.din(IN_CARGO_CHECKPOINT):
            raise RuntimeError("CARGO_NOT_CONFIRMED: pickup checkpoint is false")
        self.cargo_expected = True
        self.check_process_guards()
        print(f"CARGO CHECK OK: mode={self.guard_mode.value}")

    def end_loaded_transit(self) -> None:
        self.check_process_guards()
        self.cargo_expected = False
